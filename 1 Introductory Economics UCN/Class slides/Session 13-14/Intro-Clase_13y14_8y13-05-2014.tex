\section{Introducción}
\begin{frame}
\frametitle{Introducción}
\begin{itemize}
\item <1> Hasta ahora estudiamos la oferta y la demanda desde el punto de vista científico, ahora vamos a estudiar como la política económica puede afectar los resultados de un mercado.\\
\bigskip
\item <1> Si bien las condiciones de equilibrio pueden ser eficientes, puede que no todos estén satisfechos con dicho equilibrio.\\
\bigskip
\item <1> Para esto los economistas usan la teoría para desarrollar medidas de política económica.\\
\bigskip
\begin{itemize}
\item <1> Fijación de precios mínimos
\item <1> Fijación de precios máximos
\item <1> Impuestos $\Longrightarrow$ para que sirven?
\end{itemize}
\end{itemize}
\end{frame}



\section{Los controles de precios}
\begin{frame}
\frametitle{Los controles de precios}
\begin{itemize}
\item <1> Se utilizan cuando los diseñadores de política creen que el precio de mercado es injusto para compradores o para vendedores
\bigskip
\item <1> El gobierno entonces decide fijar precios máximos o precios mínimos.
\begin{itemize}
\bigskip
\item <1> Precio máximo $\Longrightarrow$ Fijar el precio máximo al que se puede vender un bien.
\bigskip
\item <1> Precio mínimo $\Longrightarrow$ Fijar el precio mínimo al que se puede vender un bien.
\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Cómo los precios máximos afectan los resultados del mercado?}
\begin{itemize}
\item <1> El precio máximo no es efectivo si se fija por encima del precio de equilibrio.
\bigskip
\item <1> El precio máximo es efectivo si se fija por debajo del precio de equilibrio $\Longrightarrow$ escasez
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cómo los precios máximos afectan los resultados del mercado?}
\begin{itemize}
\item <1> Un precio máximo efectivo ocasiona:
\bigskip
\begin{itemize}
\bigskip
\item <1> Escasez $\Longrightarrow$ $Q_{D}>Q_{S}$.
\begin{itemize}
\bigskip
\item <1> Escasez de gasolina de los años 70.
\end{itemize}
\bigskip
\item <1> Racionamiento no relacionado con los precios
\begin{itemize}
\bigskip
\item <1> Largas colas
\bigskip
\item <1> Discriminación basada en sesgo del vendedor
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Caso de estudio 1: colas en las gasolineras}
\begin{itemize}
\item <1> En 1973, la OPEP subió el precio del crudo en los mercados mundiales. El petróleo crudo es el principal insumo en la gasolina, por lo que los altos precios del petróleo redujo el suministro de gasolina.
\bigskip
\item <1> ¿Quién era el responsable de las largas colas en las gasolineras?
\bigskip
\item <1> Los economistas culpan a las regulaciones gubernamentales que limitan a las compañías petroleras los precios que podrían cobrar por la gasolina.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cómo los precios mínimos afectan los resultados del mercado?}
\begin{itemize}
\item <1> Pueden haber dos resultados posibles:
\begin{itemize}
\bigskip
\item <1> El precio mínimo no es efectivo si se fija por debajo del precio de equilibrio.
\bigskip
\item <1> El precio mínimo es efectivo si se fija por encima del precio de equilibrio, dando lugar a un superávit.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cómo los precios mínimos afectan los resultados del mercado?}
\begin{itemize}
\item <1> Un precio mínimo impide la oferta y la demanda se desplace hacia el precio y cantidad de equilibrio.
\bigskip
\item <1> Cuando el precio de mercado baja hasta el precio mínimo,  no puede bajar más, y por lo tanto el precio de mercado es igual al precio mínimo.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cómo los precios mínimos afectan los resultados del mercado?}
\begin{itemize}
\item <1> Un precio mínimo efectivo debe ocasionar:
\begin{itemize}
\item <1> Un superávit dado que $Q_{S}>Q_{D}$.
\bigskip
\item <1> Se podría aplicar un racionamiento ajeno a los precios basado en criterios de discriminación.
\begin{itemize}
\bigskip
\item <1> Ejemplos: salario mínimo, apoyo a los precios agrícolas
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Los impuestos}
\begin{itemize}
\item <1> Los gobiernos imponen impuestos para gastar en bienes públicos (defensa, parques, carreteras, entre otros).
\bigskip
\item <1> ¿Cómo los impuestos afectan los resultados de mercado?
\begin{itemize}
\bigskip
\item <1> Los impuestos desalientan la actividad del mercado.
\bigskip
\item <1> Cuando un bien es gravado con un impuesto la cantidad vendida es menor.
\bigskip
\item <1> Compradores y vendedores comparten la carga tributaria (incidencia del impuesto).
\bigskip
\item <1> Los impuestos llevan a un cambio en el equilibrio del mercado.
\end{itemize}
\end{itemize}
\end{frame}
