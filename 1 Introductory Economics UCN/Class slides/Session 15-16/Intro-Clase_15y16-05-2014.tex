\section{Introducción}
\begin{frame}
\frametitle{Introducción}
\begin{itemize}
\item <1> ¿El precio y cantidad de equilibrio  maximizan el bienestar total de los compradores y vendedores?\\
\bigskip
\item <1> El equilibrio del mercado refleja la forma en que los mercados asignaran los recursos escasos.\\
\bigskip
\item <1> La economía del bienestar aborda al pregunta ¿Es la asignación de mercado deseable?\\
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{La economía del bienestar}
\begin{itemize}
\item <1> La economía del bienestar es el estudio de cómo la asignación de recursos afecta al bienestar económico.
\bigskip
\item <1> Los compradores y los vendedores reciben los beneficios de la participación en el mercado.
\bigskip
\item <1> El equilibrio en un mercado maximiza el bienestar total de los compradores y vendedores.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Economía del bienestar}
\begin{itemize}
\item <1> Excedente del consumidor $\Longrightarrow$ mide el bienestar económico del lado del comprador.
\bigskip
\item <1> Excedente del productor $\Longrightarrow$ mide el bienestar económico del lado del vendedor.
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Excedente del consumidor}
\begin{itemize}
\item <1> Disponibilidad a pagar $\Longrightarrow$ máxima cantidad que un comprador pagará por un bien.
\bigskip
\item <1> Mide cómo el comprador valora el bien o servicio.
\bigskip
\item <1> El área bajo la curva de demanda y arriba del precio mide el excedente del consumidor en el mercado.
\end{itemize}
\end{frame}

\begin{itemize}

\bigskip
\item <1> Mide el beneficio para los vendedores que participan en un mercado.
\bigskip
\item <1> El área encima de la curva de oferta y debajo del precio mide el excedente del productor en el mercado.
\end{itemize}

\begin{frame}
\frametitle{Eficiencia del mercado}
\begin{itemize}
\item <1> Después de obtener los excedentes, una pregunta interesante sería ¿Es la asignación de libro mercado deseable?
\bigskip
\item <1> Excedente del consumidor:
\begin{itemize}
\item <1> Valor para los compradores $-$ cantidad pagada por los compradores.
\end{itemize}
\item <1> Excedente del productor:
\begin{itemize}
\item <1> valor para los vendedores $-$ el costo de los vendedores
\end{itemize}
\item <1> Excedente total:
\begin{itemize}
\item <1> Excedente del consumidor $+$ excedente del productor
\bigskip
\item <1> Valor para los compradores $-$ costo para los vendedores
\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Evaluación del equilibrio de mercado}
\begin{itemize}
\item <1> La eficiencia $\Longrightarrow$ propiedad de una asignación de recursos de maximizar el excedente total recibido por todos los miembros de la sociedad.
\bigskip
\item <1> La equidad $\Longrightarrow$ Justicia de la distribución del bienestar entre compradores y vendedores.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Algunas conclusiones de los resultados del mercado}
\begin{itemize}
\item <1> Pueden el planificador social alterando el reparto de consumo o el reparto de producción? $\Longrightarrow$ No
\begin{itemize}
\item <1> Los libres mercados asignan la oferta de bienes a los compradores que conceden más valor, representado por su disposición a pagar.
\item <1> Los libres mercados asignan la demanda de bienes a los vendedores que los pueden producir con el menor coste.
\end{itemize}
\item <1> Pueden el planificador social aumentando o reduciendo la cantidad del bien? $\Longrightarrow$ No
\begin{itemize}
\item <1> Los libres mercados producen la cantidad de bienes que maximiza la suma del excedente del consumidor y del productor.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Algunas conclusiones de los resultados del mercado}
\begin{itemize}
\item <1> Debido a que la asignación de mercado es eficiente, el planificador social no debe intervenir.
\bigskip
\item <1> ''Dejar las cosas como van'' $\Longrightarrow$ ''laissez faire''
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Cuando la estructura del mercado puede llevar a ineficiencia?}
\begin{itemize}
\item <1> Si un sistema de mercado no es perfectamente competitivo, puede surgir  el poder de mercado.
\begin{itemize}
\bigskip
\item <1> Capacidad para ejercer influencia en los precios.
\bigskip
\item <1> El poder de mercado puede llevar a ineficiencia en el mercado.
\end{itemize}
\bigskip
\item <1> Externalidades
\end{itemize}
\end{frame}

\frametitle{Efecto de los impuestos en el bienestar}
\begin{itemize}
\item <1> Con anterioridad observamos que un impuesto disminuye tanto el bienestar de los compradores (sube el precio que pagan) como el de los vendedores (disminuye el precio que reciben).
\bigskip
\item <1> Para mirar el efecto sobre el bienestar, debemos tener en cuenta el efecto sobre los ingresos que recauda el estado.
\end{itemize}
