\begin{frame}
\frametitle{Introducción teoría del consumidor}
\begin{itemize}
\item <1> Los consumidores eligen la \textbf{"mejor"} cesta de bienes que \textbf{"pueden"} adquirir.\\
\bigskip
\item <1> Por \textbf{"mejor"} entendemos que los consumidores más desean, y por \textbf{"pueden"} nos referimos a la que efectivamente pueden permitirse adquirir.
\bigskip
\item <1> Sea la cesta de consumo del individuo $(x_{1}, x_{2})$, con precios $(p_{1}, p_{2})$, el conjunto presupuestario se define como:
$$p_{1}x_{1}+p_{2}x_{2}\leq m$$.\\
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Propiedades del conjunto presupuestario}
\begin{itemize}
\item <1> La Recta presupuestaria es el conjunto de cestas que cuestan exactamente m:
$$p_{1}x_{1}+p_{2}x_{2}=m$$
\bigskip
\item <2> La recta presupuestaria es diferente del conjunto presupuestario.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Propiedades del conjunto presupuestario}
\begin{itemize}
\item <1> Cómo se interpretarían los interceptos $\frac{m}{p_{1}}$ y $\frac{m}{p_{2}}$?
\bigskip
\item <1> La pendiente? $-\frac{p_{1}}{p_{2}}$ $\Longrightarrow$ Mide la relación en que el mercado está dispuesto a substituir el bien 2 por el 1 (Coste de oportunidad)
\bigskip
\item <2> Ejemplo: si el consumidor va aumentar su consumo del bien 1 en $\Delta x_{1}$. ¿Cuánto tendrá que modificar su consumo del bien 2 ($\Delta x_{2}$) para satisfacer su restricción presupuestaria?
    $$p_{1}x_{1}+p_{2}x_{2}=m$$
    $$p_{1}(x_{1}+\Delta x_{1})+p_{2}(x_{2}+\Delta x_{2})=m$$
    $$p_{1}\Delta x_{1}+p_{2}\Delta x_{2}=0$$
    $$\frac{\Delta x_{2}}{\Delta x_{1}}=-\frac{p_{1}}{p_{2}}$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{La elección del consumidor}
\begin{itemize}
\item <1> Anteriormente vimos que los individuos seleccionaban las cestas que estaban a su ''alcance''.\\
\bigskip
\item <1> Ahora vamos a ver cómo el individuo selecciona las ''mejores cestas'' $\Longrightarrow$ las cestas que prefieren.\\
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{La utilidad o satisfacción}
\begin{itemize}
\item <1> Las personas escogen los bienes y servicios que valoran más $\Longrightarrow$ Cómo cuantificamos esa valoración? $\Longrightarrow$ concepto de utilidad\\
\bigskip
\item <1> El problema de cuantificar la utilidad $\Longrightarrow$ Dio paso a un enfoque de preferencias.\\
\bigskip
\item <1> Las preferencias son el fundamento para analizar la elección, la utilidad es una forma de describirlas.\\
\bigskip
\item <1> Utilidad ordinal vs utilidad cardinal.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Supuestos sobre las preferencias del consumidor en el enfoque ordinal}
\begin{itemize}
\item <1> Completitud $\Longrightarrow$ completa información
\bigskip
\item <1> Jerarquización $\Longrightarrow$ es capaz de ordenar sus preferencias
\bigskip
\item <1> Transitividad $\Longrightarrow$ es consistente en sus elecciones
\bigskip
\item <1> Insaciabilidad $\Longrightarrow$ prefiere más a menos
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Utilidad y la ley de la utilidad marginal decreciente}
\begin{itemize}
\item <1> Utilidad marginal $\Longrightarrow$ utilidad adicional que genera el consumo adicional de un bien.
\bigskip
\item <1> Porqué debería ser decreciente la utilidad marginal que genera el consumo de un bien?
\bigskip
\item <1> La utilidad total de consumir cierta cantidad de un bien es igual a la suma de las utilidades marginales hasta ese punto:
$$UT=\sum Umg$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Propiedades de las curvas de indiferencia}
\begin{itemize}
\item <1> Se prefieren las curvas de indiferencia más altas a las más bajas
\bigskip
\item <1> Las curvas de indiferencia tienen pendiente negativa (RMS negativa)
\bigskip
\item <1> Las curvas de indiferencia no se cortan
\bigskip
\item <1> Las curvas de indiferencia son combadas hacia adentro (convexas) $\Longrightarrow$ RMS decreciente
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Principio de equimarginalidad}
\begin{itemize}
\item <1> ''un consumidor con un ingreso fijo enfrentado a precios de mercado dados para los bienes logrará maximizar su satisfacción o utilidad cuando la utilidad marginal del último dólar que gaste en cada bien sea exactamente la misma que la utilidad marginal del último dólar que gaste en cualquier otro bien.''
$$\frac{Umg_{1}}{P_{1}}=\frac{Umg_{2}}{P_{2}}$$
\end{itemize}
\end{frame}
