\section{Determinantes elasticidad precio de la demanda}
\begin{frame}
\frametitle{Determinantes elasticidad precio de la demanda}
\begin{itemize}
\item <1> Elasticidad precio de la demanda, mide la respuesta de la cantidad de demandada a un cambio en los precios.\\
\begin{itemize}
\item <1> Elastica $\Longrightarrow$ respuesta grande $(>1)$
\item <1> Inelastica $\Longrightarrow$ respuesta baja $(<1)$
\item <1> Unitaria $\Longrightarrow$ respuesta igual $(=1)$
\end{itemize}
\item <1> Determinantes:\\
\begin{itemize}
\item <1> Bienes necesarios frente a bienes de lujo
\begin{itemize}
\item <1> Bienes necesarios $\Longrightarrow$ demanda inelástica
\item <1> Bienes de lujo $\Longrightarrow$ demanda elástica
\end{itemize}
\item <1> Existencia de substitutos cercanos
\begin{itemize}
\item <1> Bienes con muchos substitutos $\Longrightarrow$ demanda elástica
\item <1> Bienes con pocos substitutos $\Longrightarrow$ demanda inelástica
\end{itemize}
\item <1> Definición de mercado
\item <1> Horizonte temporal
\begin{itemize}
\item <1> Alto horizonte temporal $\Longrightarrow$ demanda elástica (ejemplo: gasolina)
\item <1> Bajo horizonte temporal $\Longrightarrow$ demanda inelástica
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cálculo de la elasticidad precio de la demanda}
\begin{itemize}
\item <1> $EPD=\frac{Var \% Q }{Var \% P}$
\bigskip
\item <2> Interpretaciones para la EPD:
\begin{itemize}
\item <2> Si $EPD(>1)$ la demanda del bien es elástica
\item <2> Si $EPD(<1)$ la demanda del bien es inelástica
\item <2> Si $EPD(=1)$ la demanda del bien es unitaria
\end{itemize}
\bigskip

\item <3> Se acuerda del ejemplo de cómo disminuir el consumo de tabaco $\Longrightarrow$ un $\Delta$ en $10\%$ del precio reduce la cantidad demandada en 4 \%.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Método del punto medio (elasticidad arco)
}
\begin{itemize}
\item <1> El problema de calcular la elasticidad entre dos puntos de una curva de demanda.
\bigskip
\item <2> Suponga que tenemos dos puntos en la curva de demanda:
$$A=(Q,P)=(120,4)$$
$$B=(Q,P)=(80, 6)$$
Realice el cálculo de la EPD de A a B y luego de B a A. ¿Les dio igual?
\item <3> El método del punto medio permite resolver dicho problema $\Longrightarrow$. Si se tienen dos puntos $(Q_{1}, P_{1})$ y $(Q_{2}, P_{2})$
$$EPD=\frac{\frac{Q_{2}-Q_{1}}{(\frac{Q_{2}+Q_{1}}{2})}}{\frac{P_{2}-P_{1}}{(\frac{P_{2}+P_{1}}{2})}}$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Las curvas de demanda y la elasticidad}
\begin{itemize}
\item <1> Entre más plana la curva de demanda, mayor es la EPD.
\bigskip
\item <2> Entre más inclinada sea la curva de demanda, menor es la EPD.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Reglas generales}
\begin{itemize}
\item <1> Si la curva de demanda es inelástica
\bigskip
\begin{itemize}
\item <1> $(EPD<1)$, un $\Delta P$ $\Longrightarrow$ $\Delta IT$
\item <1> $(EPD<1)$, un $\nabla P$ $\Longrightarrow$ $\nabla IT$
\end{itemize}
\bigskip
\item <1> Si la curva de demanda es elástica
\bigskip
\begin{itemize}
\item <1> $(EPD>1)$, un $\Delta P$ $\Longrightarrow$ $\nabla IT$
\item <1> $(EPD>1)$, un $\nabla P$ $\Longrightarrow$ $\Delta IT$
\end{itemize}
\end{itemize}
\end{frame}
