\section{Funcionamiento de los mercados}
\begin{frame}
\frametitle{Cómo describir el funcionamiento de los mercados?}
\begin{itemize}
\item <1> Se utilizan modelos, que deben:
\begin{itemize}
\item <1> Representar un fenómeno económico real,
\bigskip
\item <1> Que la representación sea simplificada, y
\bigskip
\item <1> Que se haga en forma matemática
\end{itemize}
\bigskip
\item <1> Existen Buenos y malos modelos?
\begin{itemize}
\item <1> Planteamiento directo $\Longrightarrow$ validez de los supuestos
\bigskip
\item <1> Planteamiento indirecto $\Longrightarrow$ Que pueda predecir la realidad
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ejemplo: El Modelo de Maximización de beneficios (Supuestos)}
\begin{itemize}
\item <1> El objetivo es la maximización de los beneficios.
\bigskip
\item <1> La empresa dispone de información sobre los costes.
\bigskip
\item <1> Conoce la naturaleza del mercado en el que vende sus productos.
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Ejemplo: Contrastación supuestos}
\begin{itemize}
\item <1> El objetivo de un empresario es en realidad  maximizar los beneficios?
\bigskip
\item <1> El empresario conoce realmente la naturaleza del mercado en el que vende sus productos?
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Ejemplo: Contrastación de las predicciones}
\begin{itemize}
\item <1> Idea de Milton Friedman: No importa si los supuestos son adecuados.
\bigskip
\item <1> Lo importante es si el modelo es capaz de explicar y predecir los acontecimientos del mundo real.
\bigskip
\item <1> Ejemplo:  jugar pool (billar).
\end{itemize}
\end{frame}


\section{Elementos de un modelo}
\begin{frame}
\frametitle{Elementos de un modelo en economía}
\small
\begin{itemize}
\item <1> Supuesto de ceteris-paribus (\texttt{NICHO2007})
\begin{itemize}
\item <1> Número limitado de variables (Parsimonia).
\item <1> Ejemplo: producción de Trigo.
\item <1> Ciencias sociales vs ciencias naturales.
\end{itemize}
\item <1> Todos los agentes económicos intentan optimizar algo (\texttt{NICHO2007 y Var2008})
\begin{itemize}
\item <1> Las personas intentan maximizar su bienestar.
\item <1> Empresas intentan minimizar los costes.
\item <1> Legisladores intentan maximizar el bienestar público.
\end{itemize}
\item <1> Economía positiva vs economía normativa (\texttt{NICHO2007})
\begin{itemize}
\item <1> La industria minera en Chile \textbf{gasta} 100 millones de dolares al año en gastos de capital, trabajo, y en la compensación del
    daño ambiental (las cosas como son) $\Longrightarrow$ Sólo se asume una posición de descripción y de predicción.
\item <1> La industria minera en Chile \textbf{debería gastar} más dinero en la compensación por el daño ambiental $\Longrightarrow$ Se asume
    una posición moral.
\end{itemize}
\end{itemize}
\end{frame}

\section{Teoría económica sobre el Valor}
\begin{frame}
	\frametitle{Teoría económica sobre el Valor}
	\scriptsize
	\begin{itemize}
		\item <1> Pensamiento inicial
		\begin{itemize}
			\scriptsize
			\item <1> Valor (importancia, esencialidad) 
			\item <1> Santo Tomás de Aquino: valor (Dios) y precio (humanos)
		\end{itemize}
		\item <1> Nacimiento de la economía moderna
		\begin{itemize}
			\scriptsize
			\item <1> Planteamiento científico a las cuestiones económicos.
			\item <1> La Riqueza de las Naciones (Smith, 1776). Distinguía entre Valor de uso (valor) y valor de cambio(precio).
			\item <1> Paradoja de los diamantes y el agua.
		\end{itemize}
		\item <1> La teoría del trabajo del valor de cambio
		\begin{itemize}
			\scriptsize
			\item <1> El valor de cambio de un bien viene dado por lo que cuesta producirlos (trabajo).
			\item <1> Y cual es el papel de la demanda?
		\end{itemize}
		\item <1> La revolución marginalista (1850-1880)
		\begin{itemize}
			\scriptsize
			\item <1>No es la utilidad total la que determina su valor de cambio, sino la utilidad de la última unidad consumida.
		\end{itemize}
	\item <1> La síntesis marshalliana de la oferta y la demanda
	\begin{itemize}
		\scriptsize
		\item <1> Ofrecía un marco adecuado para conciliar lo que determina el precio de un determinado bien.
	\end{itemize}
	\end{itemize}
\end{frame}
