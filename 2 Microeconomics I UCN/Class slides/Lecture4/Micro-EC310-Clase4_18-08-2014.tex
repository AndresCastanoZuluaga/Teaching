\section{Introducción teoría del consumidor}
\begin{frame}
\frametitle{Introducción teoría del consumidor}
\begin{itemize}
\item <1> Los consumidores eligen la \textbf{"mejor"} cesta de bienes que \textbf{"pueden"} adquirir.\\
\bigskip
\item <1> Qué restringe las elecciones del consumidor? $\Rightarrow$ Su presupuesto, el tiempo, entre otras.\\
\bigskip
\item <1> Cuando se puede decir que una canasta $(x_{1},...,x_{n})$ es factible a los precios $p_{1},...,p_{n}$?
\bigskip
\item <1> Cuando $p_{1}x_{1}+...+p_{n}x_{n}\leq m$. Siendo m el ingreso disponible del consumidor.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Propiedades del conjunto presupuestario}
\begin{itemize}
\item <1> Las canastas exactamente factibles representan la restricción presupuestaria del consumidor:
$$\{(x_{1},...,x_{n}) | x_{1} \geq 0,...,x_{n}\geq 0 \, ,y, \, 
 p_{1}x_{1}+p_{n}x_{n}=m\}$$
\bigskip
\item <2> El conjunto presupuestario del consumidor es el conjunto de todas las canastas factibles:
$$\{(x_{1},...,x_{n}) | x_{1} \geq 0,...,x_{n}\geq 0 \, ,y, \, 
p_{1}x_{1}+p_{n}x_{n} \leq m\}$$
\bigskip
\item <2> Por lo cual la recta presupuestaria es diferente del conjunto presupuestario. La restricción presupuestaria es la frontera superior del conjunto presupuestario. 
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Propiedades del conjunto presupuestario}
\begin{itemize}
\item <1> Cómo se interpretarían los interceptos $\frac{m}{p_{1}}$ y $\frac{m}{p_{2}}$?
\bigskip
\item <2> Cómo se interpretaría la pendiente? $-\frac{p_{1}}{p_{2}}$
\begin{itemize}
\bigskip
\item <2> Mide la relación en que el mercado está dispuesto a substituir el bien 2 por el 1.
\bigskip
\item <2> Nos dice que si incrementamos $x_{1}$ en una unidad, debemos reducir $x_{2}$ en $\frac{p_{1}}{p_{2}}$.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Pendiente de la restricción presupuestaria}
\begin{itemize}
\item <1> Mide la relación en que el mercado está dispuesto a substituir el bien 2 por el 1
\bigskip
\item <2> Ejemplo: si el consumidor va aumentar su consumo del bien 1 en $\Delta x_{1}$. ¿Cuánto tendrá que modificar su consumo del bien 2 ($\Delta x_{2}$) para satisfacer su restricción presupuestaria?
    $$p_{1}x_{1}+p_{2}x_{2}=m$$
    $$p_{1}(x_{1}+\Delta x_{1})+p_{2}(x_{2}+\Delta x_{2})=m$$
    $$p_{1}\Delta x_{1}+p_{2}\Delta x_{2}=0$$
    $$\frac{\Delta x_{2}}{\Delta x_{1}}=-\frac{p_{1}}{p_{2}}$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Análisis de casos alternos}
\begin{itemize}
\item <1> ¿Qué ocurre cuando sube el precio $p_{1}$, y  m y $p_{2}$ se mantienen fijos?
\bigskip
\item <2> ¿Qué ocurre cuando sube el precio $p_{1}$, y  m y $p_{2}$ se mantienen fijos, y el consumidor gasta toda su renta en el bien 2?
\bigskip
\item <3> ¿Qué ocurre con la recta presupuestaria cuando hay una variación simultanea (se duplican) de $p_{1}$ y $p_{2}$ ?
\bigskip
\item <4> ¿Qué ocurre con la recta presupuestaria cuando hay una variación simultanea (se duplican) de $p_{1}$, $p_{2}$ y m ?
\bigskip
\item <5> ¿Qué ocurre con la recta presupuestaria cuando hay un aumento de tanto  $p_{1}$, $p_{2}$ y disminuye m ?
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Efecto de los impuestos y las subvenciones}
\begin{itemize}
\item <1> Impuestos sobre la cantidad $\Longrightarrow$ mismo efecto que un aumento en los precios ($p_{1}+t$).
\bigskip
\item <2> Impuestos sobre el valor o ad valorem (IVA) $\Longrightarrow$ $(1+\gamma)p_{1}$
\bigskip
\item <3> Subvención a la cantidad $\Longrightarrow$ El estado da un cantidad de dinero al consumidor que depende de la cantidad que compre del bien. ($p_{1}-\pi$)
\bigskip
\item <4> Subvención ad valorem $\Longrightarrow$ depende del precio del bien $(1-\gamma)p_{1}$
\bigskip
\item <5> Impuesto o Subvención fija $\Longrightarrow$ Desplaza la restricción presupuestaria hacia adentro o hacia afuera
\bigskip
\item <6> Racionamiento $\Longrightarrow$ Establecer la cantidad máxima que se puede consumir de un bien
\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Efecto de los impuestos y las subvenciones}

		$$p_{1}x_{1}+p_{2}x_{2}=m$$ \, a
		$$(1+t)p_{1}x_{1}+(1+t)p_{2}x_{2}=m$$
		\bigskip
	\end{itemize}
\end{frame}
