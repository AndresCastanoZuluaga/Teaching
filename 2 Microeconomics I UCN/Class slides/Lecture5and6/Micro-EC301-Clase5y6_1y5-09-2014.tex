\section{Introducción teoría del consumidor}
\begin{frame}
\frametitle{Introducción teoría del consumidor}
\begin{itemize}
\item <1> En el capítulo de restricción presupuestaria vimos que los individuos seleccionaban las cestas que estaban a su ''alcance''.\\
\bigskip
\item <1> Ahora vamos a ver cómo el individuo selecciona las ''mejores cestas''.\\
\bigskip
\item <1> Un bien consumido en dos lugares o circunstancias distintas equivale a dos o más bienes distintos.
\bigskip
\item <1> Un consumidor siempre escoge la cesta más preferida dentro de las cestas que son factibles.
\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Las preferencias del consumidor: enfoque cardinal vs ordinal}
	\begin{itemize}
		\item <1> Las personas escogen los bienes y servicios que valoran más $\Longrightarrow$ Cómo cuantificamos esa valoración? $\Longrightarrow$ concepto de utilidad\\
		\bigskip
		\item <1> El problema de cuantificar la utilidad $\Longrightarrow$ Dio paso a un enfoque de preferencias.\\
		\bigskip
		\item <1> Las preferencias son el fundamento para analizar la elección, la utilidad es una forma de describirlas.\\
		\bigskip
		\item <1> Utilidad ordinal vs utilidad cardinal.
	\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Relaciones de preferencia}
\begin{itemize}
\item <1> Dadas dos cestas de consumo cualesquiera, $(x_{1}, x_{2})$ y $(y_{1},y_{2})$, el consumidor puede ordenarlas según su atractivo.
\bigskip
\item <1> Preferencia estricta ($\succ$).
\bigskip
\item <1> Indiferencia ($\sim$).
\bigskip
\item <1> Preferencia débil ($\succeq$).
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Relaciones entre las preferencias}
\begin{itemize}
\item <1> si $(x_{1}, x_{2})$ $\succeq$ $(y_{1},y_{2})$ y $(y_{1}, y_{2})$ $\succeq$ $(x_{1},x_{2})$, se concluye que $(x_{1}, x_{2})$ $\sim$ $(y_{1},y_{2})$.
\bigskip
\item <2> si $(x_{1}, x_{2})$ $\succeq$ $(y_{1},y_{2})$, pero sabemos que no se da $(x_{1}, x_{2})$ $\sim$ $(y_{1},y_{2})$, se puede concluir que
$(x_{1}, x_{2})$ $\succ$ $(y_{1},y_{2})$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Supuestos sobre las preferencias: los axiomas de la teoría del consumidor}
\begin{itemize}
\item <1> Completas $\Longrightarrow$ es posible comparar dos cestas cualesquiera 
$$(x_{1}, x_{2}) \succeq (y_{1},y_{2}) $$
$$(y_{1},y_{2})  \succeq (x_{1}, x_{2})$$
\item <2> Reflexivas $\Longrightarrow$ Cualquier cesta es al menos tan buena como ella misma $(x_{1}, x_{2})$ $\succeq$ $(x_{1},x_{2})$
\bigskip
\item <3> Transitivas $\Longrightarrow$ si $(x_{1}, x_{2})$ $\succeq$ $(y_{1},y_{2})$ y $(y_{1}, y_{2})$ $\succeq$ $(z_{1},z_{2})$, entonces $(x_{1}, x_{2})$ $\succeq$ $(z_{1},z_{2})$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Las preferencias regulares}
\begin{itemize}
\item <1> Preferencias monótonas. Si $(x_{1}, x_{2})$ es una cesta bienes y  $(y_{1},y_{2})$  es una que contiene al menos la misma cantidad de ambos bienes y más de uno de ellos entonces:
$$(y_{1},y_{2}) \succeq (x_{1}, x_{2})$$ 

$\Longrightarrow$ pendiente negativa

\bigskip
\item <2> Son convexas al origen
\bigskip
\item <3> Son densas
\bigskip
\item <4> No pueden cortarse (Listo)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Las preferencias regulares $\Longrightarrow$ conjunto convexo}
\begin{itemize}
\item <1> Si se prefieren las medias a los extremos, es decir si tenemos: $(x_{1}, x_{2})$ y  $(y_{1},y_{2})$ en la misma curva de indiferencia y tomanos una media ponderada $(\frac{1}{2}x_{1} + \frac{1}{2} y_{1} , \frac{1}{2}x_{2}+ \frac{1}{2}y_{2})$, entonces esta cesta es tan buena $(\succeq)$ o mejor $(\succ)$.
\bigskip
\item <2> De modo general: $(tx_{1}+(1-t)y_{1}, tx_{2}+(1-t)y_{2})$ $\succeq$ $(x_{1}, x_{2})$ para cualquier t tal que $(0 \leq t \leq 1)$
\bigskip
\item <3> Preferencias convexas
\bigskip
\item <4> Convexidad estricta $\Longrightarrow$ desigualdad estricta
\end{itemize}
\end{frame}
