\begin{itemize}

\item <1> De acuerdo a la restricción presupuestaria del individuo, el individuo trata de ubicarse en la curva de indiferencia más alta posible.\\

\begin{frame}
\frametitle{La elección óptima}
\begin{itemize}
\item <1> La condición de tangencia se convierte en una \emph{Condición Suficiente} sólo en el caso de que las preferencias sean convexas.
\bigskip
\item <1> Si las curvas son estrictamente convexas, sólo habrá una elección óptima en cada restricción presupuestaria.
\bigskip
\item <1> La elección óptima interior para una curva de indiferencia de preferencias regulares satisface dos condiciones:
\begin{itemize}
	\item <1> El ingreso se agota: $p_{1}x_{1}^{*}+p_{1}x_{1}^{*}=m$
	\item <1> $RMS=-\frac{p_{1}}{p_{2}}$
\end{itemize}
\item <1> Qué implicaciones económicas tienen que RMS debe ser igual a la pendiente de la restricción presupuestaria en un óptimo interior?
\begin{itemize}
\item <1> $RMS=-\frac{p_{1}}{p_{2}}$
\item <1> Imagine que la RMS es distinta de la relación de precios ¿Está tomando el consumidor una decisión óptima?
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{La demanda del consumidor}
\begin{itemize}
\item <1> La elección óptima de los bienes 1 y 2 $(x_{1}^{*}, x_{2}^{*})$, dado un conjunto de precios y renta determinado, se denomina cesta demandada.
\bigskip
\item <1> La mejor de las canastas factibles es conocida como la demanda ordinaria a los precios y ingresos dados.
\bigskip
\item <1> Funciones de demanda.
$$x_{1}(p_{1}, p_{2}, m)$$
$$x_{2}(p_{1}, p_{2}, m)$$
\bigskip
\item <1> Cada tipo de preferencia da lugar a funciones de demanda distintas.
\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Ejemplo: estimando la demanda ordinaria para el caso de una función Cobb-Douglas}
	\begin{itemize}
		\item <1> Si las preferencias son Cobb-Douglas:
		$$U(x_{1}, x_{2})=x_{1}^{a}x_{2}^{b}$$
		\item <2> Cuyas utilidades marginales son:
		$$UMg_{1}=\frac{\partial U(x_{1}, x_{2})}{\partial x_{1}}=ax_{1}^{a-1}x_{2}^{b}$$
		$$UMg_{2}=\frac{\partial U(x_{1}, x_{2})}{\partial x_{2}}=bx_{1}^{a}x_{2}^{b-1}$$
		\item <3> Y la tasa marginal de sustitución:
		$$TMgS=RMS=-\frac{\frac{\partial U(x_{1}, x_{2})}{\partial x_{1}}}{\frac{\partial U(x_{1}, x_{2})}{\partial x_{2}}}=-\frac{ax_{2}}{bx_{1}}$$	
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Ejemplo: estimando la demanda ordinaria para el caso de una función Cobb-Douglas}
	\begin{itemize}
		\item <1> En $(x_{1}^{*}, x_{2}^{*})$ se debe cumplir que TMgS (RMS)  es igual a la pendiente de la restricción presupuestaria, por lo tanto:
		$$TMgS=-\frac{p_{1}}{p_{2}}$$
      $$-\frac{ax_{2}^{*}}{bx_{1}^{*}}=-\frac{p_{1}}{p_{2}}$$
		$$x_{2}^{*}=\frac{bp_{1}}{ap_{2}}x_{1}^{*}$$
		\item <1> Y además hemos visto que  $(x_{1}^{*}, x_{2}^{*})$ agota el presupuesto:
	 $$p_{1}x_{1}^{*}+p_{1}x_{1}^{*}=m$$
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Ejemplo: estimando la demanda ordinaria para el caso de una función Cobb-Douglas}
	\begin{itemize}
		\item <1> Sustituyendo $x_{2}^{*}$ en la condición 2 se obtiene:
		$$x_{1}^{*}=\frac{am}{(a+b)p_{1}}$$
		\item <1> Sustituyendo $x_{1}^{*}$ en la solución para  $x_{2}^{*}$ se obtiene:
		$$x_{2}^{*}=\frac{bm}{(a+b)p_{2}}$$
		\item <1> Por lo tanto hemos descubierto que la mejor canasta factible para el consumidor con preferencias Cobb-Douglas es:
$$(x_{1}^{*},x_{2}^{*})=
\{\frac{am}{(a+b)p_{1}} , \frac{bm}{(a+b)p_{2}}\}$$
		\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ejemplo soluciones de esquina $\Longrightarrow$sustitutos perfectos}
\begin{itemize}
\item <1> En definitiva, si la función de utilidad es $U(x_{1},x_{2})=x_{1}+x_{2}$ la canasta óptima es $(x_{1}^{*},x_{2}^{*})$ donde:
\bigskip
\begin{itemize}
\item <1> Si $p_{1}<p_{2}$:
$$(x_{1}^{*},x_{2}^{*}) =(\frac{m}{p_{1}}, 0)$$
\bigskip
\item <1>Si $p_{1}>p_{2}$:
$$(x_{1}^{*},x_{2}^{*}) =(0,\frac{m}{p_{2}})$$
\end{itemize}
\bigskip
\item <1> Conclusión: si dos bienes son sustitutos perfectos, el consumidor comprará el más barato.
\end{itemize}
\end{frame}
